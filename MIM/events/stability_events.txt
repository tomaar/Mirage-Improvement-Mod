#low stability selector event, hidden
country_event = {
	id = stability.3
	title = stability.3.t	
	desc = stability.3.desc	#
	picture = GFX_					# 
	
	hidden = yes

	trigger = {
		has_war = yes
		has_civil_war = no
		date > 1939.9.1
		OR = {
			has_stability < 0.5
			has_war_support < 0.5
		}
		NOT = {
			AND = {
				is_major = yes
				all_enemy_country = {
					is_major = no
				}
			}
		}
		NOT = { 
			has_country_flag = crisis_in_progress
		}
		NOT = { # Don't fire stability events if the country is not mobilising at all
			OR = {
				has_idea = disarmed_nation
				has_idea = volunteer_only
			}
			OR = {
				has_idea = undisturbed_isolation
				has_idea = isolation
				has_idea = civilian_economy
			}
		}
	}

	mean_time_to_happen = { 
		days = 360 
		modifier = {
			factor = 2.0
			has_idea = disarmed_nation
		}
		modifier = {
			factor = 1.5
			has_idea = volunteer_only
		}
		modifier = {
			factor = 2.5
			has_idea = undisturbed_isolation
		}
		modifier = {
			factor = 2.0
			has_idea = isolation
		}
		modifier = {
			factor = 1.5
			has_idea = civilian_economy
		}
		modifier = {
			factor = 2
			NOT = { has_government = democratic }
		}
		modifier = {
			factor = 0.75
			OR = {
				has_stability < 0.4
				has_war_support < 0.4
			}
		}
		modifier = {
			factor = 0.75
			OR = {
				has_stability < 0.2
				has_war_support < 0.2
			}
		}
	}

	immediate = { 
		random_list = {
			20 = { 
				modifier = {
					factor = 0
					has_country_flag = draft_dodging_resolved
				}
				modifier = {
					factor = 0
					has_idea = disarmed_nation
				}
				modifier = {
					factor = 0
					has_idea = volunteer_only
				}
				modifier = {
					factor = 0.1
					has_war_support > 0.5
				}
				add_ideas = draft_dodging 
				country_event = stability.21
			} #draft dodging
			20 = { 
				modifier = {
					factor = 0
					has_country_flag = strikes_resolved
				}
				modifier = {
					factor = 0.1
					has_stability > 0.5
				}
				add_ideas = strikes_1 
				country_event = stability.22
			} #strikes
			20 = { 
				modifier = {
					factor = 0
					has_country_flag = mutiny_resolved
				}
				modifier = {
					factor = 0.1
					has_war_support > 0.5
				}
				add_ideas = mutiny_1 
				country_event = stability.23
			} #mutinies
		}
		if = {
			limit = {
				NOT = {
					has_country_flag = crisis_in_progress
				}
			}
			set_country_flag = crisis_in_progress
		}
	}
	
	option = { 
		name = stability.3.a
		ai_chance = {
			
		}
		
	}

}
